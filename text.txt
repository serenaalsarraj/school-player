<script>

// PASSCODE SYSTEM
function checkPass(){
  if(document.getElementById("pass").value === "0000"){
    document.getElementById("lockscreen").style.display = "none";
  } else {
    alert("Wrong Passcode!");
  }
}

// MUSIC CONTROL
function toggleMusic(){
  const m = document.getElementById("music");
  if(m.paused){ m.play(); } else { m.pause(); }
}

// YOUTUBE SEARCH
function searchYT(){
  let q = document.getElementById("search").value;
  window.open("https://www.youtube.com/results?search_query=" + encodeURIComponent(q));
}

// LOAD YOUTUBE VIDEO
function loadYT(){
  let id = document.getElementById("yt").value.trim();
  document.getElementById("player").innerHTML =
    `<iframe id="vid" src="https://www.youtube.com/embed/${id}" allowfullscreen></iframe>`;
}

// LOAD TIKTOK
function loadTT(){
  let url = document.getElementById("tt").value.trim();
  document.getElementById("player").innerHTML =
    `<blockquote class="tiktok-embed" cite="${url}"><a href="${url}"></a></blockquote>`;
}

// FULLSCREEN
function goFullscreen(){
  let v = document.getElementById("vid");
  if(v){ v.requestFullscreen(); }
}

// NOTES SAVE
document.getElementById("notes").value = localStorage.getItem("notes") || "";
function saveNotes(){
  localStorage.setItem("notes", document.getElementById("notes").value);
  alert("Notes saved!");
}
</script>
